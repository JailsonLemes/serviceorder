Sistema de Disputas

Mecanismo de resolução de conflitos entre empresa e freelancer. Escrow congelado durante a disputa. Admin com SLA de 48h e autoridade total sobre resolução financeira e reputacional.

1. Quando Pode Ser Aberta

A disputa é último recurso — só fica disponível depois que as partes tiveram oportunidade de resolver diretamente. Regra fundamental: se existe pagamento capturado em escrow, a disputa pode ser aberta.

Status da OrdemEmpresaFreelancerObservação
IN_PROGRESS✓✓Situação mais comum. Escrow capturado.
PENDING_REVIEW✓✓Empresa não aprova e não quer solicitar ajuste.
ADJUSTMENT_REQUESTED✗✓ (mín 2 ciclos)Proteção ao freelancer contra ciclos abusivos de revisão.
DRAFT / PUBLISHED / IN_APPLICATION / SELECTED / ACCEPTED✗✗Escrow não capturado. Cancelamento resolve.
APPROVED / COMPLETED / CANCELLED✗✗Estados terminais. Sem disputa possível.

⚠️ Unicidade: apenas uma disputa ativa por ordem. Se a disputa for encerrada com REOPEN e o problema persistir, uma nova pode ser aberta — com prioridade máxima e atribuição ao mesmo admin.

Evidências de Entrega

  • Descrições submetidas pelo freelancer
  • Evidências enviadas na disputa por ambas as partes
Perfil das Partes
  • Score de reputação atual
  • Histórico de disputas (ganhas/perdidas)
  • Taxa de conclusão, cancelamentos

Sinais de Risco (auto)
  • Freelancer com 3+ disputas perdidas em 90d
  • Empresa com padrão serial de disputas
  • Ciclos excessivos de ajuste (abuso)
  • Sem nenhuma evidência de entrega submetida

src/dispute/dispute.service.ts — open()
static async open(input: OpenDisputeInput) {
  const order   = await OrderRepo.findById(input.orderId);
  const payment = await PaymentRepo.findByOrder(input.orderId);

  // Guard 1: status da ordem
  const DISPUTABLE = ['IN_PROGRESS', 'PENDING_REVIEW', 'ADJUSTMENT_REQUESTED'];
  if (!DISPUTABLE.includes(order.status))
    throw new InvalidStateError('ORDER_NOT_DISPUTABLE');

  // Guard 2: freelancer só pode disputar ADJUSTMENT_REQUESTED com ≥ 2 ciclos
  if (input.actorRole === 'FREELANCER' && order.status === 'ADJUSTMENT_REQUESTED') {
    const cycles = await OrderStatusRepo.countAdjustmentCycles(input.orderId);
    if (cycles < 2) throw new BusinessError('MIN_ADJUSTMENT_CYCLES_NOT_REACHED');
  }

  // Guard 3: descrição mínima para evitar disputas frívolas
  if (input.description.length < 50)
    throw new ValidationError('DESCRIPTION_TOO_SHORT');

  // Guard 4: unicidade — sem disputa ativa na mesma ordem
  const active = await db.query(
    `SELECT id FROM disputes WHERE order_id=$1 AND status NOT IN ('CLOSED')`, [input.orderId]
  );
  if (active.rows.length) throw new ConflictError('DISPUTE_ALREADY_ACTIVE');

  return db.transaction(async (trx) => {
    // 1. Cria disputa
    const dispute = await trx.query(`
      INSERT INTO disputes (order_id, payment_id, opened_by, opened_by_role,
        reason_category, description, status, expected_resolution_at)
      VALUES ($1,$2,$3,$4,$5,$6,'OPEN', NOW() + INTERVAL '72 hours')
      RETURNING *
    `, [input.orderId, payment.id, input.actorId,
       input.actorRole, input.reason, input.description]);

    // 2. Congela escrow — job de liberação ignora status DISPUTED
    await trx.query(`UPDATE payments SET status='DISPUTED' WHERE id=$1`, [payment.id]);

    // 3. Muda status da ordem
    await trx.query(`UPDATE orders SET status='DISPUTED' WHERE id=$1`, [input.orderId]);

    // 4. Registra no histórico de status (imutável)
    await trx.query(`
      INSERT INTO order_status_history (order_id, from_status, to_status, actor_id, actor_role)
      VALUES ($1,$2,'DISPUTED',$3,$4)
    `, [input.orderId, order.status, input.actorId, input.actorRole]);

    // 5. Evento assíncrono: notificações + alerta admin on-call
    await queue.publish('dispute.opened', { disputeId: dispute.rows[0].id });
    return dispute.rows[0];
  });
}
src/dispute/dispute.service.ts — resolve()
static async resolve(input: ResolveDisputeInput) {
  const dispute = await DisputeRepo.findById(input.disputeId);
  const payment = await PaymentRepo.findById(dispute.payment_id);

  // Apenas admin atribuído resolve (ou sênior com flag)
  if (dispute.assigned_to !== input.adminId && !input.isSeniorOverride)
    throw new ForbiddenError('NOT_ASSIGNED_ADMIN');

  if (input.resolutionNotes.length < 50)
    throw new ValidationError('RESOLUTION_NOTES_REQUIRED');

  const split = this.calcSplit(input.decision, input.companyRefundAmount, payment);

  return db.transaction(async (trx) => {
    // 1. Executa financeiro — estorno e/ou transferência na mesma transação
    if (split) await DisputePaymentExecutor.execute(trx, payment, split);

    // 2. Fecha disputa
    await trx.query(`
      UPDATE disputes SET status=$1, resolution=$2, resolution_amount=$3,
        resolution_notes=$4, resolved_by=$5, resolved_at=NOW()
      WHERE id=$6
    `, [resolvedStatus(input.decision), input.decision, split?.companyRefund,
       input.resolutionNotes, input.adminId, input.disputeId]);

    // 3. Muda status da ordem conforme decisão
    const orderStatus = {
      REFUND_FULL:      'CANCELLED',
      REFUND_PARTIAL:   'CANCELLED',
      PAYMENT_RELEASED: 'COMPLETED',
      SPLIT:            'CANCELLED',
      NO_ACTION:        'APPROVED',   // retoma fluxo normal de liberação
      REOPEN:           'IN_PROGRESS',
    }[input.decision];
    await trx.query(`UPDATE orders SET status=$1 WHERE id=$2`, [orderStatus, dispute.order_id]);

    // 4. REOPEN: redefine prazo obrigatório + reabre chat
    if (input.decision === 'REOPEN') {
      if (!input.newDeadlineAt) throw new ValidationError('NEW_DEADLINE_REQUIRED');
      await trx.query(`UPDATE orders SET deadline_at=$1 WHERE id=$2`, [input.newDeadlineAt, dispute.order_id]);
      await trx.query(`UPDATE chat_channels SET status='ACTIVE', closed_at=NULL WHERE order_id=$1`, [dispute.order_id]);
    }

    // 5. Penalidades de reputação + suspensão aplicadas assincronamente
    await queue.publish('dispute.resolved', {
      disputeId:   input.disputeId,    decision:     input.decision,
      openedByRole: dispute.opened_by_role,
      companyId:   dispute.company_user_id,
      freelancerId: dispute.freelancer_user_id,
      split,
    });
  });
}

private static calcSplit(decision: string, companyRefund: string | undefined, payment: Payment) {
  switch (decision) {
    case 'REFUND_FULL':
      return { companyRefund: payment.gross_amount, freelancerPay: '0', platformRetains: '0' };
    case 'PAYMENT_RELEASED':
      return { companyRefund: '0', freelancerPay: payment.freelancer_amount, platformRetains: payment.platform_fee_amount };
    case 'REFUND_PARTIAL':
    case 'SPLIT': {
      const r  = Decimal.sub(payment.gross_amount, companyRefund ?? '0');
      const fee= Decimal.mul(r, payment.platform_fee_rate);
      return { companyRefund, freelancerPay: Decimal.sub(r,fee).toFixed(2), platformRetains: fee.toFixed(2) };
    }
    default: return null;  // NO_ACTION e REOPEN: sem movimento financeiro imediato
  }
}

8. Monitoramento de SLA

src/dispute/jobs/dispute-sla.job.ts — cron: */15 * * * *
export async function disputeSLAJob() {
  const now = new Date();

  // 1. OPEN sem atribuição em 24h → auto-escala
  const unassigned = await db.query(`
    SELECT id FROM disputes
    WHERE status='OPEN' AND assigned_to IS NULL
      AND created_at + INTERVAL '24 hours' <= $1
    FOR UPDATE SKIP LOCKED LIMIT 20
  `, [now]);
  for (const d of unassigned.rows) {
    await db.query(`UPDATE disputes SET status='ESCALATED' WHERE id=$1`, [d.id]);
    await alertSeniorAdmin({ disputeId: d.id, reason: 'SLA_NO_ASSIGNMENT', channel: 'pagerduty' });
  }

  // 2. UNDER_REVIEW com SLA de resolução vencido → alerta supervisor
  const overdue = await db.query(`
    SELECT id, assigned_to FROM disputes
    WHERE status='UNDER_REVIEW' AND expected_resolution_at < $1
    FOR UPDATE SKIP LOCKED LIMIT 20
  `, [now]);
  for (const d of overdue.rows) {
    // Não escala automaticamente: admin pode ter boa razão
    // Supervisor decide se intervém ou estende o SLA
    await alertSupervisor({ disputeId: d.id, adminId: d.assigned_to });
  }

  // 3. AWAITING_EVIDENCE sem resposta em 48h → volta a UNDER_REVIEW
  const evidenceExpired = await db.query(`
    SELECT d.id, d.assigned_to FROM disputes d
    WHERE d.status = 'AWAITING_EVIDENCE'
      AND d.evidence_requested_at + INTERVAL '48 hours' <= $1
      AND NOT EXISTS (
        SELECT 1 FROM dispute_evidences de
        WHERE de.dispute_id = d.id AND de.created_at > d.evidence_requested_at
          AND de.submitted_by_role != 'ADMIN'
      )
    FOR UPDATE SKIP LOCKED LIMIT 20
  `, [now]);
  for (const d of evidenceExpired.rows) {
    await db.query(`UPDATE disputes SET status='UNDER_REVIEW' WHERE id=$1`, [d.id]);
    await alertAdmin(d.assigned_to, { reason: 'EVIDENCE_EXPIRED_DECIDE_WITHOUT' });
  }
}

9. Suspensão Automática por Reincidência

Usuário com 3 ou mais disputas perdidas em 90 dias é suspenso automaticamente e entra em fila de revisão humana. A suspensão bloqueia candidaturas (freelancer) ou publicação de novas ordens (empresa) até a revisão.

src/dispute/services/suspension.service.ts
export async function checkSuspension(userId: string, role: string) {
  const lostCount = await db.query(`
    SELECT COUNT(*) AS n FROM disputes d
    WHERE d.created_at > NOW() - INTERVAL '90 days'
      AND d.status LIKE 'RESOLVED_%'
      AND (
        (d.opened_by = $1 AND d.resolution IN ('REFUND_FULL','REFUND_PARTIAL','PAYMENT_RELEASED'))
        -- abriu e perdeu qualquer coisa
      )
  `, [userId]);

  if (Number(lostCount.rows[0].n) >= 3) {
    await db.query(`
      UPDATE users SET status='SUSPENDED', suspended_at=NOW(),
        suspension_reason='AUTO_DISPUTE_REINCIDENCE'
      WHERE id=$1
    `, [userId]);
    await alertComplianceTeam({ userId, role, reason: '3_DISPUTES_LOST_90D' });
    logger.warn({ userId, role }, 'Usuário suspenso por reincidência em disputas');
  }
}
