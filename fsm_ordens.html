Implementação da FSM (TypeScript)

type OrderStatus = 'DRAFT' | 'PUBLISHED' | 'IN_APPLICATION' | 'SELECTED'
  | 'ACCEPTED' | 'IN_PROGRESS' | 'PENDING_REVIEW' | 'ADJUSTMENT_REQUESTED'
  | 'APPROVED' | 'COMPLETED' | 'DISPUTED' | 'CANCELLED';

type ActorRole = 'EMPRESA' | 'FREELANCER' | 'SISTEMA' | 'ADMIN';

interface Transition {
  from: OrderStatus;
  to: OrderStatus;
  allowedActors: ActorRole[];
  guard: (order: Order, actor: User) => boolean | Promise<boolean>;
  onTransition: (order: Order, actor: User, ctx: TransitionContext) => Promise<void>;
}

const TRANSITIONS: Transition[] = [
  {
    from: 'DRAFT', to: 'PUBLISHED',
    allowedActors: ['EMPRESA'],
    guard: (o) => !!o.title && o.title.length >= 10
              && !!o.description && o.description.length >= 50
              && !!o.deadline_at && o.deadline_at > addHours(new Date(), 24),
    onTransition: async (o, _, ctx) => {
      await ctx.eventBus.emit('order.published', { orderId: o.id });
      await ctx.searchIndex.index(o); // Elasticsearch
    }
  },
  {
    from: 'SELECTED', to: 'ACCEPTED',
    allowedActors: ['FREELANCER'],
    guard: async (o, actor) => {
      if (o.selected_freelancer_id !== actor.freelancer_id) return false;
      const payment = await ctx.paymentRepo.findByOrder(o.id);
      return payment?.status === 'AUTHORIZED'; // escrow pré-autorizado
    },
    onTransition: async (o, _, ctx) => {
      await ctx.chatService.createChannel(o.id);   // libera chat
      await ctx.paymentService.capture(o.id);       // captura escrow
      await ctx.eventBus.emit('order.accepted', { orderId: o.id });
    }
  },
  {
    from: 'APPROVED', to: 'COMPLETED',
    allowedActors: ['SISTEMA'],
    guard: async (o) => {
      const daysSinceApproval = diffInDays(new Date(), o.approved_at!);
      return daysSinceApproval >= ESCROW_RELEASE_DAYS;
    },
    onTransition: async (o, _, ctx) => {
      await ctx.paymentService.release(o.id);
      await ctx.ratingService.openWindow(o.id); // abre janela 72h
      await ctx.reputationService.scheduleUpdate(o);
      await ctx.eventBus.emit('order.completed', { orderId: o.id });
    }
  }
  // ... demais transições
];

async function transition(
  orderId: string,
  toStatus: OrderStatus,
  actor: User,
  reason?: string
): Promise<void> {
  const order = await orderRepo.findById(orderId);
  const rule = TRANSITIONS.find(t => t.from === order.status && t.to === toStatus);

  if (!rule) throw new InvalidTransitionError(order.status, toStatus);
  if (!rule.allowedActors.includes(actor.role)) throw new ForbiddenTransitionError();
  if (!await rule.guard(order, actor)) throw new TransitionGuardError();

  await db.transaction(async (trx) => {
    await trx('orders').update({ status: toStatus }).where({ id: orderId });
    await trx('order_status_history').insert({
      order_id: orderId, from_status: order.status, to_status: toStatus,
      actor_id: actor.id, actor_role: actor.role, reason
    });
    await rule.onTransition(order, actor, { trx, ...services });
  });
}
